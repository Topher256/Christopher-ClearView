"""
video_processor.py — Orchestrates the full processing pipeline.
Ties together temporal analysis, summarization, event storage,
thumbnail generation, and database updates.
"""

import os
import time
import logging
import cv2

from utils.temporal_analysis import analyse_video
from utils.summarizer         import summarize_video
from utils.database           import (
    update_video_meta, update_summary, save_events, log_action
)

logger = logging.getLogger(__name__)

THUMB_DIR = os.path.join(os.path.dirname(os.path.dirname(__file__)), 'static', 'img', 'thumbs')
os.makedirs(THUMB_DIR, exist_ok=True)


def extract_thumbnail(video_path: str, video_id: int) -> str:
    """Capture a frame from the middle of the video and save as JPEG."""
    cap = cv2.VideoCapture(video_path)
    if not cap.isOpened():
        return ''
    total = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
    cap.set(cv2.CAP_PROP_POS_FRAMES, total // 2)
    ret, frame = cap.read()
    cap.release()
    if not ret:
        return ''
    thumb_path = os.path.join(THUMB_DIR, f'thumb_{video_id}.jpg')
    cv2.imwrite(thumb_path, cv2.resize(frame, (320, 180)))
    return f'img/thumbs/thumb_{video_id}.jpg'


def process_video(
    video_id:    int,
    summary_id:  int,
    user_id:     int,
    input_path:  str,
    output_path: str,
    compression: int,
) -> dict:
    """
    Full processing pipeline:
    1. Temporal analysis
    2. Update video metadata in DB
    3. Generate summarized video
    4. Store events in DB
    5. Update summary record
    6. Return result dict
    """
    t0 = time.time()

    try:
        log_action(user_id, video_id, 'process_start', 'Temporal analysis started')

        # Step 1 — Temporal analysis
        logger.info(f"[video {video_id}] Starting temporal analysis")
        analysis = analyse_video(input_path)

        # Step 2 — Update video metadata
        update_video_meta(
            video_id,
            analysis['duration'],
            analysis['fps'],
            analysis['width'],
            analysis['height'],
        )

        # Step 3 — Summarize
        logger.info(f"[video {video_id}] Generating summary at {compression}% compression")
        stats = summarize_video(input_path, output_path, analysis, compression)

        # Step 4 — Save events
        if analysis['events']:
            save_events(summary_id, analysis['events'])

        # Step 5 — Update summary record
        update_summary(
            summary_id,
            stats['original_dur'],
            stats['summary_dur'],
            len(analysis['events']),
            round(time.time() - t0, 2),
            analysis['motion_avg'],
            status='done',
        )

        log_action(user_id, video_id, 'process_done',
                   f"Summary done: {stats['summary_dur']:.1f}s / {stats['original_dur']:.1f}s")

        # Step 6 — Thumbnail
        extract_thumbnail(input_path, video_id)

        return {
            'success':  True,
            'analysis': analysis,
            'stats':    stats,
        }

    except Exception as exc:
        logger.exception(f"[video {video_id}] Processing failed: {exc}")
        update_summary(summary_id, 0, 0, 0, 0, 0, status='error')
        log_action(user_id, video_id, 'process_error', str(exc))
        return {'success': False, 'error': str(exc)}
